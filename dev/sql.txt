CREATE TABLE prescriptions (
    id INT PRIMARY KEY AUTO_INCREMENT,
    patient_id INT,
    prescription_text TEXT,
    prescribing_date DATE,
    prescribing_doctor VARCHAR(255),
    FOREIGN KEY (patient_id) REFERENCES patients(id)
);
